ingredients:
  - "2 frozen bananas"
  - "1 cup spinach"
  - "1 tbsp matcha green tea powder"
  - "1/2 cup coconut milk"
  - "1/4 cup granola"
  - "1 kiwi (sliced)"
  - "2 tbsp coconut flakes"
  - "1 tbsp chia seeds"
instructions:
  - "Blend frozen bananas, spinach, matcha powder, and coconut milk until smooth."
  - "Pour into bowls."
  - "Top with granola, kiwi slices, coconut flakes, and chia seeds."
